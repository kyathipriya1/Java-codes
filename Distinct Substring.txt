
import java.util.*;
import java.lang.*;
import java.io.*;

class Codechef
{
	public static void main (String[] args) throws java.lang.Exception
	{
		// your code goes here
        Scanner sc=new Scanner(System.in);
        int t=sc.nextInt();
        while(t-->0){
            int x=sc.nextInt();
            int y=sc.nextInt();
            if(canForm(x,y)){
                System.out.println("YES");
            }else{
                System.out.println("NO");
            }
        }
	}
	private static boolean canForm(int x,int y){
	    int min=(y*(y+3))/2-1;
	    return x>=min;
	}
}
